68void Camera::TrackMovement(ui::Window& window, const float delta_time) {
69 glm::vec3 front = glm::normalize(this->look_at() - this->position());
70 glm::vec3 right = glm::normalize(glm::cross(front, this->up()));
71 front = glm::normalize(glm::cross(this->up(), right));
72 glm::vec3 position_change = glm::vec3(0.0);
73 if (window.GetKey(GLFW_KEY_A) == GLFW_PRESS) {
74 position_change -= this->translation_speed() * delta_time * right;
75 }
76 if (window.GetKey(GLFW_KEY_S) == GLFW_PRESS) {
77 position_change -= this->translation_speed() * delta_time * front;
78 }
79 if (window.GetKey(GLFW_KEY_D) == GLFW_PRESS) {
80 position_change += this->translation_speed() * delta_time * right;
81 }
82 if (window.GetKey(GLFW_KEY_W) == GLFW_PRESS) {
83 position_change += this->translation_speed() * delta_time * front;
84 }
85 if (window.GetKey(GLFW_KEY_E) == GLFW_PRESS) {
86 position_change += this->translation_speed() * delta_time * this->up();
87 }
88 if (window.GetKey(GLFW_KEY_Q) == GLFW_PRESS) {
89 position_change -= this->translation_speed() * delta_time * this->up();
90 }
91 this->set_position(this->position() + position_change);
92 this->set_look_at(this->look_at() + position_change);
93}

95void Camera::TrackRotation(ui::Window& window, double x, double y) {
96 static constexpr float kEps = 1e-6;
97 y = -y;
98 if (glm::epsilonEqual(glm::length(this->last_cursor_pos_), 0.0f, kEps)) {
99 this->last_cursor_pos_ = {x, y};
100 }
101
102 if (window.GetKey(GLFW_KEY_LEFT_ALT) == GLFW_PRESS ||
103 window.GetKey(GLFW_KEY_RIGHT_ALT) == GLFW_PRESS) {
104 auto&& front = glm::normalize(this->look_at() - this->position());
105 auto&& euler = common::math::VectorToEuler(front);
106 float yaw = euler[0];
107 float pitch = euler[1];
108 yaw -= this->yaw_speed_ * (x - this->last_cursor_pos_.x);
109 pitch += this->pitch_speed_ * (y - this->last_cursor_pos_.y);
110 pitch = glm::clamp(pitch, /*minVal=*/glm::radians(/*degrees=*/-89.0f),
111 /*maxVal=*/glm::radians(/*degrees=*/+89.0f));
112 front = common::math::EulerToVector(glm::vec2(yaw, pitch));
113 this->set_look_at(this->position() + front);
114 }
115 this->last_cursor_pos_ = {x, y};
116}
